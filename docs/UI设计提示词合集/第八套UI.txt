# Superdesign Portfolio

A premium portfolio showcase component featuring a dark mode aesthetic, large typography, and dynamic motion effects.

## Features
- **Immersive Dark Theme**: Deep black backgrounds with high-contrast text.
- **Custom Typography**: Uses 'Satoshi' variable font for a modern look.
- **Motion Effects**: Scroll-triggered reveals, parallax hero, and hover interactions using Framer Motion.
- **Responsive Layout**: Fully responsive grid and typography.

## Usage

```tsx
import { SuperdesignPortfolio } from '@/sd-components/a36ad1fa-6ed0-4a22-b24c-c64c863d9a71';

function Page() {
  return <SuperdesignPortfolio />;
}
```

## Props
This component currently does not accept any props as it is a self-contained page layout. To customize content, modify the `Component.tsx` file directly.

const ProjectCard = ({ title, category, image, index }: { title: string, category: string, image: string, index: number }) => {
  const [isHovered, setIsHovered] = useState(false);

  return (
    <motion.div 
      initial={{ opacity: 0, y: 50 }}
      whileInView={{ opacity: 1, y: 0 }}
      viewport={{ once: true }}
      transition={{ duration: 0.6, delay: index * 0.1 }}
      className="group relative cursor-pointer"
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
    >
      <div className="aspect-[4/3] overflow-hidden bg-[#111111]">
        <motion.img 
          src={image} 
          alt={title}
          className="w-full h-full object-cover opacity-80 group-hover:opacity-100 transition-opacity duration-500"
          animate={{ scale: isHovered ? 1.05 : 1 }}
          transition={{ duration: 0.6 }}
        />
      </div>
      <div className="mt-6 flex justify-between items-start border-t border-[#333333] pt-6">
        <div>
          <h3 className="text-2xl md:text-3xl font-bold mb-1 group-hover:text-[#FF6B50] transition-colors">{title}</h3>
          <p className="text-[#666666] text-sm uppercase tracking-wider">{category}</p>
        </div>
        <motion.div 
          animate={{ x: isHovered ? 5 : 0, opacity: isHovered ? 1 : 0.5 }}
